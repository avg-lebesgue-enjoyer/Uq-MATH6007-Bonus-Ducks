\begin{frame}{The idea} % .;
  
  \par Fix $X \in \Top_{*}$\footnote{Basepoint always named $*$ and often omitted.}.
  \only<1-1>{
    \par The fundamental group $\pi_1 (X)$ analyses maps of the form
    \begin{align*}
      \I &\longrightarrow X \\
      \partial \I &\longrightarrow \set*{*}
    \end{align*}
    up to homotopy.
  }
  \only<2-2>{
    \par The homotopy groups $\pi_{\attentive{n}} (X)$ analyse maps of the form
    \begin{align*}
      \I^{\attentive{n}} &\longrightarrow X \\
      \partial \I^{\attentive{n}} &\longrightarrow \set*{*}
    \end{align*}
    up to homotopy.
  }

\end{frame}
